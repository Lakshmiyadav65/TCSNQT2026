# Lexicographical (Dictionary) sorting in Python # Python's default tuple sorting is lexicographical. # It compares elements from left to right. pairs_data = [(10, 4), (3, 2), (5, 2), (3, 1), (10, 5)] # Method 1: Default behavior for tuples (X asc, then Y asc) # This is the standard lexicographical sort. sorted_default = sorted(pairs_data) print("Default (X asc, Y asc):", sorted_default) # Method 2: Custom sort (e.g., X asc, then Y desc) # We use a lambda function to define a custom key. # For descending order, we can negate numerical values. sorted_custom = sorted(pairs_data, key=lambda p: (p[0], -p[1])) print("Custom (X asc, Y desc):", sorted_custom) # Example of sorting by Y first, then X (both ascending) sorted_y_first = sorted(pairs_data, key=lambda p: (p[1], p[0])) print("Y first (Y asc, X asc):", sorted_y_first)

Problem 02: Fraud Transactions

💡 PROBLEM LOGIC & APPROACH

Problem: Identify fraud if the same Sender, Receiver, and Amount occur within 60 seconds.

Format: Sender, Receiver, Amount, Timestamp

Logic: Use a HashMap where:
• Key = (Sender, Receiver, Amount)
• Value = Last Timestamp
🐍 FULL PYTHON SOLUTION
# Efficient Fraud Detection using a Dictionary (HashMap) # Time Complexity: O(N) where N is the number of transactions. # Space Complexity: O(M) where M is the number of unique (Sender, Receiver, Amount) combinations. def detect_fraudulent_transactions(transactions): # last_seen_time stores the timestamp of the last occurrence for each unique transaction key. # Key: (sender, receiver, amount) tuple # Value: timestamp of its last occurrence last_seen_time = {} # List to store transactions identified as fraudulent fraudulent_transactions = [] for transaction_str in transactions: # Parse the transaction string into its components sender, receiver, amount_str, timestamp_str = transaction_str.split(',') # Convert amount and timestamp to appropriate types amount = int(amount_str) current_timestamp = int(timestamp_str) # Create a unique key for this transaction type transaction_key = (sender, receiver, amount) # Check if this transaction key has been seen before if transaction_key in last_seen_time: # Calculate the time difference since its last occurrence time_difference = current_timestamp - last_seen_time[transaction_key] # If the difference is 60 seconds or less, it's a fraud if time_difference <= 60: fraudulent_transactions.append(transaction_str) # Always update the last seen timestamp for the current transaction key # This ensures we always track the most recent occurrence. last_seen_time[transaction_key] = current_timestamp return fraudulent_transactions # Example Usage: sample_transactions = [ "A,B,100,10", # First occurrence of (A,B,100) "C,D,50,20", # Unique transaction "A,B,100,40", # (A,B,100) again, within 60s of timestamp 10 (40-10=30 <= 60) -> FRAUD "E,F,200,70", # Unique transaction "A,B,100,120", # (A,B,100) again, within 60s of timestamp 40 (120-40=80 > 60) -> NOT FRAUD "C,D,50,80" # (C,D,50) again, within 60s of timestamp 20 (80-20=60 <= 60) -> FRAUD ] detected_frauds = detect_fraudulent_transactions(sample_transactions) print("Detected Fraudulent Transactions:", detected_frauds) # Expected Output: ['A,B,100,40', 'C,D,50,80']

# Full Kruskal's Algorithm with Disjoint Set Union (Union-Find) # Kruskal's algorithm finds a Minimum Spanning Tree (MST) for a connected, undirected graph. # It works by sorting all edges by weight and adding them to the MST if they don't form a cycle. class UnionFind: # Initializes the DSU structure. Each element is initially its own parent. def __init__(self, num_elements): self.parent = list(range(num_elements)) # parent[i] stores the parent of element i self.rank = [0] * num_elements # Used for optimization (union by rank) def find(self, i): # Finds the representative (root) of the set containing element i. # Implements path compression for efficiency. if self.parent[i] == i: return i self.parent[i] = self.find(self.parent[i]) # Path compression return self.parent[i] def union(self, i, j): # Unites the sets containing elements i and j. # Implements union by rank for efficiency. root_i = self.find(i) root_j = self.find(j) if root_i != root_j: # Attach smaller rank tree under root of high rank tree # (Union by Rank) if self.rank[root_i] < self.rank[root_j]: self.parent[root_i] = root_j elif self.rank[root_i] > self.rank[root_j]: self.parent[root_j] = root_i else: self.parent[root_j] = root_i self.rank[root_i] += 1 return True # Union successful return False # i and j are already in the same set (would form a cycle) def kruskals_mst(num_nodes, edges): # Input: num_nodes (int), edges (list of tuples: (u, v, weight)) # Output: (total_min_cost, list_of_mst_edges) # Step 1: Sort all edges in non-decreasing order of their weights. # This is crucial for Kruskal's greedy approach. edges.sort(key=lambda edge: edge[2]) # Sort by the third element (weight) # Initialize Disjoint Set Union (DSU) structure for num_nodes nodes. dsu = UnionFind(num_nodes) min_cost = 0 mst_edges = [] edges_count = 0 # To track how many edges are added to MST # Iterate through the sorted edges. for u, v, weight in edges: # An MST for N nodes has exactly N-1 edges. if edges_count == num_nodes - 1: break # MST is complete # Check if adding this edge (u, v) forms a cycle. # If u and v are already in the same set, adding the edge creates a cycle. if dsu.find(u) != dsu.find(v): # If they are in different sets, unite them. dsu.union(u, v) min_cost += weight mst_edges.append((u, v, weight)) edges_count += 1 # Check if the graph was connected (MST should have num_nodes - 1 edges) if edges_count != num_nodes - 1 and num_nodes > 1: return float('inf'), [] # Graph is not connected, no MST possible return min_cost, mst_edges # Example Usage: # Graph with 4 nodes (0, 1, 2, 3) # Edges are (u, v, weight) graph_edges = [ (0, 1, 10), (0, 2, 6), (0, 3, 5), (1, 3, 15), (2, 3, 4) ] num_vertices = 4 total_cost, mst_result = kruskals_mst(num_vertices, graph_edges) print("Minimum Cost to Connect All Nodes:", total_cost) print("Edges in MST:", mst_result) # Expected Output: # Minimum Cost to Connect All Nodes: 19 # Edges in MST: [(2, 3, 4), (0, 3, 5), (0, 2, 6)] (order might vary slightly based on tie-breaking)
